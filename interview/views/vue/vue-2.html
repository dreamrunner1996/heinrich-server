建立一个 bus.js 文件 
    import Vue from 'vue'
    const Bus = new Vue()
    export default Bus
在 main.js 导入 
    import EventBus from '@/bus.js'
    Vue.use(EventBus)
    Vue.prototype.$EventBus = EventBus
a 文件派发一个值 
    this.$EventBus.$emit('uploading', 'sure')
b 文件接收它 
    this.$EventBus.$on('uploading', val => { console.log(val) })  //输出 sure
